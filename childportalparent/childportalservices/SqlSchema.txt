
CREATE DATABASE `childportal` /*!40100 DEFAULT CHARACTER SET utf8 */$$

CREATE TABLE `address` (
  `ADDRESS_ID` bigint(20) NOT NULL AUTO_INCREMENT,
  `CARE_OF` varchar(100) DEFAULT NULL,
  `LANDMARK` varchar(100) DEFAULT NULL,
  `LOCALITY` varchar(100) DEFAULT NULL,
  `VILLAGE_TOWN_CITY` varchar(100) DEFAULT NULL,
  `DISTRICT` varchar(100) DEFAULT NULL,
  `BUILDING` varchar(100) DEFAULT NULL,
  `STREET` varchar(100) DEFAULT NULL,
  `POST_OFFICE` varchar(100) DEFAULT NULL,
  `STATE` varchar(45) DEFAULT NULL,
  `PIN_CODE` bigint(20) DEFAULT NULL,
  PRIMARY KEY (`ADDRESS_ID`),
  UNIQUE KEY `ADDRESS_ID_UNIQUE` (`ADDRESS_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=29 DEFAULT CHARSET=utf8$$

delimiter $$

CREATE TABLE `admission` (
  `ADMISSION_ID` bigint(20) NOT NULL,
  `USER_ID` bigint(20) NOT NULL,
  `ORG_ID` bigint(20) NOT NULL,
  `ADMISSION_NO` varchar(20) NOT NULL,
  `STANDARD` varchar(20) NOT NULL,
  `CURRENT_STATUS` varchar(20) NOT NULL,
  `SCHOOLING_YEAR` varchar(20) NOT NULL,
  `CREATED_ON` datetime NOT NULL,
  `LAST_UPDATED_ON` datetime NOT NULL,
  `CREATED_BY` varchar(100) NOT NULL,
  `LAST_UPDATED_BY` varchar(100) NOT NULL,
  `COMMENTS` varchar(200) DEFAULT NULL,
  `AUDIT_COMMENTS` varchar(1000) NOT NULL,
  `ENROLLED_BY` varchar(45) NOT NULL,
  `ENROLLMENT_DATE` datetime NOT NULL,
  PRIMARY KEY (`ADMISSION_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$

delimiter $$

CREATE TABLE `organization` (
  `ORG_ID` bigint(20) NOT NULL,
  `ORG_TYPE` varchar(20) NOT NULL,
  `ORG_CODE` varchar(20) NOT NULL,
  `ORG_NAME` varchar(30) NOT NULL,
  `ADDRESS` varchar(100) NOT NULL,
  `DISTRICT` varchar(20) NOT NULL,
  `STATE` varchar(20) NOT NULL,
  PRIMARY KEY (`ORG_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$

delimiter $$

CREATE TABLE `person` (
  `USER_ID` bigint(20) NOT NULL,
  `AADHAR_NO` varchar(100) NOT NULL,
  `NAME` varchar(50) NOT NULL,
  `LOCAL_NAME` varchar(100) DEFAULT NULL,
  `FATHER_NAME` varchar(50) DEFAULT NULL,
  `MOTHER_NAME` varchar(50) DEFAULT NULL,
  `GENDER` varchar(1) NOT NULL,
  `DOB` datetime NOT NULL,
  `DOB_TYPE` varchar(45) NOT NULL,
  `ADDRESS_ID` varchar(20) NOT NULL,
  `MOBILE` varchar(20) DEFAULT NULL,
  `EMAIL` varchar(50) DEFAULT NULL,
  PRIMARY KEY (`USER_ID`,`AADHAR_NO`),
  UNIQUE KEY `ADDRESS_ID_UNIQUE` (`ADDRESS_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$

delimiter $$

CREATE TABLE `report_enrollment` (
  `id` bigint(20) NOT NULL,
  `YEAR` int(11) NOT NULL,
  `MONTH` int(11) NOT NULL,
  `DISTRICT` varchar(45) NOT NULL,
  `STATE` varchar(45) NOT NULL,
  `AADHAR_NO` varchar(45) NOT NULL,
  `ENROLLED_BY` varchar(45) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$

delimiter $$

CREATE TABLE `role` (
  `ROLE_ID` bigint(20) NOT NULL,
  `ROLE` varchar(100) NOT NULL,
  PRIMARY KEY (`ROLE_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$

delimiter $$

CREATE TABLE `stat_date_keys` (
  `ID` bigint(20) NOT NULL,
  `yearValue` int(11) NOT NULL,
  `monthValue` int(11) NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE KEY `ID_UNIQUE` (`ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$

delimiter $$

CREATE TABLE `user_role` (
  `USER_ID` bigint(20) NOT NULL,
  `ROLE_ID` bigint(20) NOT NULL,
  PRIMARY KEY (`USER_ID`,`ROLE_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$
